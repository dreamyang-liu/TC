def longest_unique(s):
    window = {}; left = 0; ans = 0
    for right, c in enumerate(s):
        window[c] = window.get(c, 0) + 1  # 1) 扩张
        while window[c] > 1:                # 2) 不合法 → 收缩
            window[s[left]] -= 1
            left += 1
        ans = max(ans, right - left + 1) # 3) 更新答案
    return ans
