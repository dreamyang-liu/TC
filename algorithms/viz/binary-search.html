def search(nums, target):
    lo, hi = 0, len(nums)         # 半开 [lo, hi)
    while lo < hi:
        mid = (lo + hi) // 2
        if nums[mid] < target:    # ← 只改这里
            lo = mid + 1           # 砍掉左半
        else:
            hi = mid                # 砍掉右半
    return lo
